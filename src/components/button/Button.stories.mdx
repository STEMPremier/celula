import { Description, Meta, Preview, Props, Story } from '@storybook/addon-docs/blocks';
import { action } from '@storybook/addon-actions';
import { boolean, select, text } from '@storybook/addon-knobs';

import Button from '.';
import {
  BUTTON_TYPES as TYPES,
  COMPONENT_STATES as STATES,
  COLORS,
  SIZES
} from '../../utils/constants';

<Meta component={Button} title="Components/Atoms/Button" />

# Button
<Description of={Button} />
<Props of={Button} />

<Story name="Button with Knobs" parameters={{ docs: { disable: true }}}>
  <Button
    className={text('className', '')}
    color={select('color', COLORS, 'primary')}
    disabled={boolean('disabled', false)}
    handleClick={action('`Button with Knobs` onClick event triggered.')}
    size={select('size', SIZES, 'small')}
    type={select('type', TYPES, 'default')}
  >
    {text('children', 'a button')}
  </Button>
</Story>

### States
<Preview>
  <Story name="Button States">
